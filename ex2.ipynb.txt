    "import numpy as np\n",
    "# Step activation function\n",
    "def step_function(net):\n",
    "    return 1 if net > 0 else 0\n",
    "# Perceptron training function\n",
    "def train_perceptron(X, y, weights, bias, learning_rate=0.1):\n",
    "    epoch = 0\n",
    "    while True:\n",
    "        print(f\"\\nEpoch {epoch + 1}\")\n",
    "        total_error = 0\n",
    "\n",
    "        for i in range(len(X)):\n",
    "            net = np.dot(X[i], weights) - bias\n",
    "            y_pred = step_function(net)\n",
    "            error = y[i] - y_pred\n",
    "            total_error += abs(error)\n",
    "            # Update rule\n",
    "            weights = weights + learning_rate * error * X[i]\n",
    "            bias = bias - learning_rate * error\n",
    "            print(f\"Input: {X[i]}, Net: {net:.2f}, \"\n",
    "                  f\"Output: {y_pred}, Target: {y[i]}, Error: {error}\")\n",
    "        print(\"Updated Weights:\", weights)\n",
    "        print(\"Updated Bias:\", bias)\n",
    "        epoch += 1\n",
    "        if total_error == 0:\n",
    "            print(\"Training converged\")\n",
    "            break\n",
    "    return weights, bias\n",
    "# Input patterns\n",
    "X = np.array([\n",
    "    [0, 0],\n",
    "    [0, 1],\n",
    "    [1, 0],\n",
    "    [1, 1]\n",
    "])\n",
    "# Outputs\n",
    "y_and = np.array([0, 0, 0, 1])\n",
    "y_or = np.array([0, 1, 1, 1])\n",
    "# AND gate initial values\n",
    "print(\"\\n--- Training AND Gate ---\")\n",
    "weights_and, bias_and = train_perceptron(\n",
    "    X, y_and,\n",
    "    weights=np.array([0.2, 0.2]),\n",
    "    bias=-0.3\n",
    ")\n",
    "# OR gate initial values\n",
    "print(\"\\n--- Training OR Gate ---\")\n",
    "weights_or, bias_or = train_perceptron(\n",
    "    X, y_or,\n",
    "    weights=np.array([0.2, 0.2]),\n",
    "    bias=-0.1\n",
    ")\n",
    "\n",
    "# Testing\n",
    "def test_perceptron(X, weights, bias):\n",
    "    print(\"\\nTesting Results:\")\n",
    "    for x in X:\n",
    "        output = step_function(np.dot(x, weights) - bias)\n",
    "        print(f\"Input: {x}, Output: {output}\")\n",
    "print(\"\\n--- Testing AND Gate ---\")\n",
    "test_perceptron(X, weights_and, bias_and)\n",
    "print(\"\\n--- Testing OR Gate ---\")\n",
    "test_perceptron(X, weights_or, bias_or)"
